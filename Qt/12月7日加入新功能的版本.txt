void FindInDegree(ALGraph G, int indegree[])
{
  int i;
  ArcNode *p;
  for(int j=0; i<N; i++)
    {
      indegree[j]=-1;
    }
  for (i=1; i<=G.vexnum; i++)
    {
      p=G.vertices[i].firstarc;
      while (p)
        {
          indegree[p->adjvex]++;
          p=p->next;
        }
    }
}
bool TopologicalSort(ALGraph G)
{
  cout<<"TopologicalSort:"<<endl;
  stack<int> S;
  int Count,k,i;
  ArcNode *p;
  int indegree[N];
  FindInDegree(G, indegree); // 对各顶点求入度
  for (i=1; i<=G.vexnum; ++i)
// 建零入度顶点栈S
    if (!indegree[i]) S.push(i);
// 入度为0者进栈
  Count = 0;
// 对输出顶点计数
  /*for(int jj=1;jj<=G.vexnum;jj++)
  cout<<indegree[jj]<<" ";
  cout<<endl;*///测试
  while (!S.empty())
    {
      i=S.top();
      S.pop();
      ans[++ans[0]]=i;
      cout<<i<<" "<<G.vertices[i].classname<<"  "<<G.vertices[i].classtime<<endl;
      ++Count;
      for (p=G.vertices[i].firstarc; p; p=p->next)
        {
          k = p->adjvex;
          if (!(--indegree[k])) S.push(k);
        }
    }
  if (Count<G.vexnum) return false;
  else return true;
} // TopologicalSort
